📘 Complete HTML Tags Guide
🔹 1. Basic Structure Tags
<!DOCTYPE html>  <!-- Defines HTML5 document -->
<html>           <!-- Root element -->
<head>           <!-- Metadata container -->
<title>          <!-- Page title -->
<body>           <!-- Visible content -->
🔹 2. Text & Formatting Tags
<h1> to <h6>     <!-- Headings -->
<p>              <!-- Paragraph -->
<br>             <!-- Line break -->
<hr>             <!-- Horizontal line -->

<b>              <!-- Bold -->
<strong>         <!-- Important text -->
<i>              <!-- Italic -->
<em>             <!-- Emphasized -->
<u>              <!-- Underline -->
<mark>           <!-- Highlight -->
<small>          <!-- Small text -->
<del>            <!-- Deleted text -->
<ins>            <!-- Inserted text -->
<sub>            <!-- Subscript -->
<sup>            <!-- Superscript -->
🔹 3. Links & Media
<a>              <!-- Hyperlink -->
<img>            <!-- Image -->
<video>          <!-- Video -->
<audio>          <!-- Audio -->
<source>         <!-- Media source -->
<iframe>         <!-- Embed content -->
🔹 4. Lists
<ul>             <!-- Unordered list -->
<ol>             <!-- Ordered list -->
<li>             <!-- List item -->
<dl>             <!-- Description list -->
<dt>             <!-- Term -->
<dd>             <!-- Description -->
🔹 5. Tables
<table>          <!-- Table -->
<tr>             <!-- Table row -->
<td>             <!-- Table data -->
<th>             <!-- Table header -->
<thead>          <!-- Table head -->
<tbody>          <!-- Table body -->
<tfoot>          <!-- Table footer -->
<caption>        <!-- Table caption -->
🔹 6. Forms & Input
<form>           <!-- Form container -->
<input>          <!-- Input field -->
<textarea>       <!-- Multi-line input -->
<button>         <!-- Button -->
<select>         <!-- Dropdown -->
<option>         <!-- Dropdown option -->
<label>          <!-- Label -->
<fieldset>       <!-- Group inputs -->
<legend>         <!-- Fieldset title -->
<datalist>       <!-- Auto suggestions -->
🔹 7. Semantic Tags (VERY IMPORTANT)
<header>         <!-- Top section -->
<nav>            <!-- Navigation -->
<main>           <!-- Main content -->
<section>        <!-- Section -->
<article>        <!-- Independent content -->
<aside>          <!-- Sidebar -->
<footer>         <!-- Bottom section -->
🔹 8. Layout & Container Tags
<div>            <!-- Block container -->
<span>           <!-- Inline container -->
🔹 9. Meta & SEO Tags
<meta>           <!-- Metadata -->
<link>           <!-- External resources -->
<style>          <!-- Internal CSS -->
<script>         <!-- JavaScript -->
🔹 10. Advanced / Useful Tags
<details>        <!-- Expandable content -->
<summary>        <!-- Title for details -->
<progress>       <!-- Progress bar -->
<meter>          <!-- Measurement -->
<time>           <!-- Date/time -->
<code>           <!-- Code snippet -->
<pre>            <!-- Preformatted text -->
🔥 Must-Know Priority (Focus First)

If you're learning, prioritize these:

Structure: html, head, body
Text: h1-h6, p
Links/Media: a, img
Lists: ul, ol, li
Tables: table, tr, td
Forms: form, input, button
Semantic: header, section, article, footer
Containers: div, span
